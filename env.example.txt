# Wevro 環境變數配置範例
# 請複製此檔案為 .env 並填入實際的值

# 資料庫配置 (必需)
DATABASE_URL=postgresql://username:password@host:port/database

# OpenAI API 配置 (必需 - 用於 AI 功能)
AI_INTEGRATIONS_OPENAI_BASE_URL=https://api.openai.com/v1
AI_INTEGRATIONS_OPENAI_API_KEY=your_openai_api_key_here

# Firebase Configuration (必需 - 用於使用者認證)
VITE_FIREBASE_API_KEY=your_firebase_api_key
VITE_FIREBASE_AUTH_DOMAIN=your-project.firebaseapp.com
VITE_FIREBASE_PROJECT_ID=your-project-id
VITE_FIREBASE_STORAGE_BUCKET=your-project.appspot.com
VITE_FIREBASE_MESSAGING_SENDER_ID=your_sender_id
VITE_FIREBASE_APP_ID=your_app_id

# Google OAuth Client ID (必需 - 用於原生 Google 登入)
# 格式：xxxxx.apps.googleusercontent.com
# 獲取方式：
#   方法 1: Firebase Console > Project Settings > General > Your apps > Web app > OAuth 2.0 Client ID
#   方法 2: Google Cloud Console > APIs & Services > Credentials > OAuth 2.0 Client IDs > Web client
# 注意：這是用於移動端原生 Google 登入，不是 Firebase API Key
VITE_GOOGLE_CLIENT_ID=your_oauth_2.0_client_id.apps.googleusercontent.com

# Firebase Admin SDK (必需 - 後端認證)
FIREBASE_SERVICE_ACCOUNT_KEY={"type":"service_account","project_id":"your-project-id",...}

# 伺服器配置 (選用，預設為 5000)
PORT=5000
NODE_ENV=development

# ===========================================
# 如何取得這些設定值：
# ===========================================
#
# 1. DATABASE_URL (Neon Database):
#    - 前往 https://neon.tech/
#    - 註冊免費帳號
#    - 建立新專案
#    - 複製連接字串
#
# 2. OpenAI API:
#    - 前往 https://platform.openai.com/
#    - 建立 API 金鑰
#    - 確保帳戶有可用額度
#
# 3. Firebase (使用者認證):
#    - 前往 https://console.firebase.google.com/
#    - 建立新專案或選擇現有專案
#    - 啟用 Authentication (Google + Email/Password)
#    - 在專案設定中取得 Web 應用程式配置
#    - 在服務帳戶中下載 Service Account Key (JSON)
#    - 詳細步驟請參考 FIREBASE_SETUP.md
#
# 4. Google OAuth Client ID (用於原生 Google 登入):
#    - 前往 Firebase Console > Project Settings > General
#    - 在 "Your apps" 部分找到 Web app
#    - 複製 "OAuth 2.0 Client ID"（格式：xxxxx.apps.googleusercontent.com）
#    - 或者前往 Google Cloud Console > APIs & Services > Credentials
#    - 在 "OAuth 2.0 Client IDs" 中找到 Web client，複製 Client ID
#    - 將此值填入 VITE_GOOGLE_CLIENT_ID
#
# 4. RevenueCat (App 內購 - 選用):
#    - 前往 https://www.revenuecat.com/
#    - 註冊帳號（每月 $2,500 收入內免費）
#    - 建立 App (iOS + Android)
#    - 在 App Store Connect 和 Play Console 設定產品
#    - 複製 API Key
#
# ===========================================
# 使用方法：
# ===========================================
#
# 1. 將此檔案複製為 .env
# 2. 填入實際的設定值
# 3. 執行 npm run dev 啟動專案

# RevenueCat 付費系統配置 (選用 - 用於 App 內購)
REVENUECAT_API_KEY=your_revenuecat_api_key_here
REVENUECAT_WEBHOOK_SECRET=your_webhook_secret_here

# App 資訊
IOS_BUNDLE_ID=com.yourcompany.wevro
ANDROID_PACKAGE_NAME=com.yourcompany.wevro
